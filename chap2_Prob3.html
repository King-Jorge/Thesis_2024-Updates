##### Parameters
c <- 1
U2max <- 50
beta <- 0.0002
theta <- 2
mu <- 0.334
T <- 1200
tstart1 <- 0
toff1 <- 1500


# This is designed to trigger termination of the simulation when I1 < 0.5
rootfun <- function(t, y, parameters){
  I1 = y[2]
  y1 = 1 - I1 
  return(y1)
}


u2fun <- function(t) {
  u2 <- parameters$u2max
  if(t < tstart1) u2 <- 0
  if(t > toff1 & t < (toff1 + 10)) u2 <- 0
  return(u2)
}


SI <- function(t, y, parameters) {
  S <- y[1]
  I1 <- y[2]
  I2 <- y[3]
  U1 <- y[4]
  iso <- y[5]
  u2 <- u2fun(t)
  
  dS = - beta*S*(I1 + c*I2) 
  dI1 = beta*S*(I1 + c*I2) - mu*I1
  dI2 = theta - 2*mu*I2 - u2*I2
  dU2 = u2*I2
  diso = u2*I2 - 2*mu*iso
  return(list(c(dS,dI1,dI2,dU2,diso)))
  
}

# Simulation for different values of u2max
u2maxvec <- seq(0, 3, 0.5)
DataProb <- NULL

for (u2max in u2maxvec) {
  parameters <- list(u2max = u2max)
  
  out1 <- ode(y = c(S = 5000, I1 = 10, I2 = theta/((2*mu) + u2max), U2 = 0, iso = 0), parms = parameters, times = seq(0,T,.05), func = SI, events = list(func = NULL, root = TRUE, terminalroot = 1), rootfun=rootfun)
  
  
  
  out1 <- data.frame(out1)
  J1 <- cumsum(beta * out1$S * (out1$I1 + c * out1$I2) * diff(c(0, out1$time)))
  u <- rep(0, length(J1))
  u[which(diff(c(out1$U2[1], out1$U2)) * diff(c(0, out1$time)) > 0.001)] <- max(c(out1$I2, out1$iso))
  
  DataProb <- rbind(DataProb, data.frame(out1, u = u, j = J1, u2max = u2max))
}
